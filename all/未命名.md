@startuml 自习室预约管理系统-管理员用例
left to right direction
skinparam actor {
    BackgroundColor #lightblue
}
skinparam usecase {
    BackgroundColor #white
    BorderColor #333333
}
skinparam rectangle {
    BorderColor black
    BorderThickness 1.5
}
skinparam note {
    BackgroundColor #fff9e6
    BorderColor #d6b656
}

' 参与者
actor "管理员" as Admin

' 系统边界
rectangle 自习室预约管理系统 {
    ' 用户管理模块
    usecase "用户管理" as admin_usermgmt
    usecase "查看用户列表" as admin_userlist
    usecase "搜索用户" as admin_usersearch
    usecase "设置用户角色" as admin_userrole
    usecase "重置用户密码" as admin_resetpwd
    usecase "禁用/启用用户" as admin_disable
    usecase "删除用户" as admin_userdel

    ' 自习室管理模块
    usecase "自习室管理" as admin_roommgmt
    usecase "新增自习室" as admin_roomadd
    usecase "编辑自习室" as admin_roomedit
    usecase "上传封面图片" as admin_roomcover
    usecase "重新生成座位布局" as admin_roomlayout
    usecase "删除自习室" as admin_roomdel

    ' 座位管理模块
    usecase "座位管理" as admin_seatmgmt
    usecase "新增座位" as admin_seatadd
    usecase "重新变换布局" as admin_seatlayout
    usecase "设置维修状态" as admin_seatmaintain
    usecase "删除座位" as admin_seatdel
    usecase "清空自习室座位" as admin_seatclear

    ' 预约管理模块
    usecase "预约管理" as admin_apptmgmt
    usecase "查看全部预约" as admin_apptlist
    usecase "按条件筛选预约" as admin_apptfilter
    usecase "强制取消预约" as admin_apptcancel
    usecase "删除预约记录" as admin_apptdel
    usecase "导出预约Excel报表" as admin_apptexport

    ' 统计分析模块
    usecase "统计分析" as admin_statsmgmt
    usecase "查看系统概览统计" as admin_statsoverview

    ' 任务管理模块
    usecase "任务管理" as admin_taskmgmt
    usecase "查看定时任务列表" as admin_tasklist
    usecase "手动触发执行任务" as admin_taskrun
}
